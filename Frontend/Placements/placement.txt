      // Function to show Schools content
      function Blogs() {
        document.getElementById('content').innerHTML = `
        <div class="container mx-auto py-8 px-4 darkable">
          <h1 class="text-3xl font-bold textskyblue mb-24 text-center darkable">Welcome to Our Blog Page</h1>

          <div class="flex flex-wrap gap-4 darkable " id="blogContainer">
            
              <!-- Blog Posts will be added here dynamically -->
          </div>
        </div>
        `;
        // Fetch data from the API and render it
        fetch('http://localhost:8000/TechNews')
          .then(response => response.json())
          .then(data => {
            const blogContainer = document.getElementById('blogContainer');

            data.data.forEach(blog => {
              const blogDiv = document.createElement('div');
              blogDiv.classList.add('border-2' , 'border-white' , 'p-4' , 'dimskyblue', 'rounded', 'p-4', 'w-40%', 'h-auto', 'max-w-48', 'max-h-full' , 'darkable');
              blogDiv.innerHTML = `
              
                        <img src="${blog.image}" alt="${blog.title}" class="w-full h-auto mb-4 rounded darkable" style="max-width: 100%; max-height: 300px;">
                        <h2 class="text-xl font-bold mb-2 darkable">${blog.title}</h2>
                        <p class="text-gray-700 mb-2 darkable">${blog.content}</p>
                        <p class="text-gray-600 font-bold darkable">Author: ${blog.author}</p>
                        
                        
                    `;


              blogContainer.appendChild(blogDiv);
            });
          })
          .catch(error => {
            console.error('Error fetching data:', error);
          });
      }

    function showTable(year) {
